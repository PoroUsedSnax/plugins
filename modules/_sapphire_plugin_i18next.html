    Edits a message using an objects option.

    since

    2.0.0

    example
    await editLocalized(message, { keys: 'commands/ping:success' });
    // ➡ "Pong!"
    example

    const took = 96;

    await editLocalized(message, {
        keys: 'commands/ping:success',
        formatOptions: { latency, took }
    });
